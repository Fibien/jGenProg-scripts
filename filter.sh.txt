#!/bin/bash

# Check if filename is provided as an argument
if [ $# -ne 2 ]; then
    echo "Need from file and to file as arguments"
fi

filename=$1

# Flag to indicate if we are currently in a section starting with "checkout"
in_checkout=false

while IFS= read -r line; do

  if  [[ "$line" == *"Checking out"*  ]] || [[ "$line" == *"Check out"*  ]]; then
    echo "$line" >> $2
  elif [[ "$line" == *"java"* ]]; then
	echo "$line" >> $2
	# Add line break
	echo "" >> $2 
  fi
done < $1
